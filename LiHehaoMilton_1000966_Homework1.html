# Creating a linearly separable data set
%matplotlib inline
import numpy as np
import matplotlib.pyplot as plt
mean = np.zeros(4)
cov = np.identity(4)
theta = np.random.multivariate_normal(mean, cov, 1).T
theta = theta/np.linalg.norm(theta)
X = np.random.multivariate_normal(mean, cov, 1000)
Y = np.sign(np.dot(X,theta))
plt.scatter(X[:,1], X[:,2],c=Y,s=20, cmap='gray')
plt.show()

# (3c) Importing the perceptron
from sklearn.linear_model import perceptron

# (3a) Implementing the perceptron algorithm
# We know that if y.theta.x is less than or equal to 0, we must change theta into a new value
# We also know that theta has 4 terms, and thus we can create a weighted classifier, c with the same size

c = np.zeros(theta.size)
notcompleted = True

# Set iteration to be 1 (1st attempt)
iteration = 1

# For each k value from 0 to 3
while notcompleted:
    
    # Set the number of mistakes to be 0 
    mistakes = 0
    
    # (3b) Runs through all samples
    for t in range(0,Y.size):
        for k in range(0,c.size):
            
            # Do a dot product y.c.x
            h = np.dot(c,X[t])
            dot = np.dot(h,Y[t])
            
            # If the dot product is less than or equal to 0, we update the c value
            if (dot <= 0):
                change = np.multiply(Y[t],X[t])
                c = np.add(c,change)
                mistakes += 1
    
    # Prints the status message
    print('Iteration ' + str(iteration) + ', ' + str(mistakes) + ' mistakes found')
    
    # Checks if the algorithm has completed
    if (mistakes > 0):
        iteration += 1
    
    else:
        # (3b) Once the algorithm has completed, we normalise and print c and compare with the original theta
        print('Converged!')
        notcompleted = False
        
        # Calculating c
        print('C: ' + str(c) + '\n')
        c = c/np.linalg.norm(c)
        print ('Normalised C: ' + str(c))
        
        # Converting theta into a list for ease of comparison
        ctheta = np.zeros(theta.size)
        for i in range(theta.size):
            ctheta[i] = theta[i][0]
        print ('Normalised Theta: ' + str(ctheta))
        
        # Show the difference between each c and theta value (using c - theta)
        print ('Difference between C and theta: ' + str(np.subtract(c,ctheta)) + '\n')
        
         # (3c) Perceptron using sci-kit
        p = perceptron.Perceptron()
        p.fit(X, Y.ravel(), coef_init=None, intercept_init=None, sample_weight=None)
        
        # (3c) Checking the accuracy of the results
        print ('Overall predictive accuracy of P: ' + str(p.score(X,Y)*100) + '%')
        
        # (3c) Normalizing P and printing them out
        p = p.coef_/np.linalg.norm(p.coef_)
        ptheta = np.zeros(theta.size)
        for i in range(theta.size):
            ptheta[i] = p[0][i]
        print('Normalized P: ' + str(ptheta))
        print('Difference between C and P: ' + str(np.subtract(c,ptheta)))
